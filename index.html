📱 ИК-ПРИЁМНИК VS1838 · 7 ИНЖЕНЕРНЫХ ЭКСПЕРИМЕНТОВ

Принцип работы: инфракрасный пульт модулирует сигнал частотой 38 кГц. Приёмник VS1838 (TSOP38238) демодулирует его и выдаёт цифровой сигнал, который Arduino расшифровывает с помощью библиотеки IRremote .

📡 КАК РАБОТАЕТ ИК-ПУЛЬТ

🔹 Протокол NEC

Самый распространённый протокол. Каждое нажатие передаёт 32 бита: адрес (8 бит), инвертированный адрес (8 бит), команда (8 бит), инвертированная команда (8 бит) .

A5 5A 01 FE

🔹 Частота 38 кГц

Все бытовые пульты используют модуляцию 38 кГц (иногда 36, 40, 56 кГц). Это позволяет отсекать помехи от освещения и других источников .

VS1838 настроен именно на 38 кГц.

🔌 БАЗОВОЕ ПОДКЛЮЧЕНИЕ (все эксперименты)
VCC → 5V (питание)
GND → GND (земля)
OUT → D11 (сигнал)
ℹ️ ПРИМЕЧАНИЕ: У разных моделей ИК-приёмников распиновка может отличаться! Обычно:
Если смотреть на приёмник выпуклой стороной к себе: слева — OUT, посередине — GND, справа — VCC.
Всегда сверяйтесь с даташитом или маркировкой на корпусе .

// ЭКСПЕРИМЕНТ 1: Чтение кодов ИК-пульта // Установите библиотеку IRremote через менеджер библиотек #include #define IR_PIN 11 IRrecv irrecv(IR_PIN); // Создаём объект для приёма decode_results results; // Структура для хранения результата void setup() { Serial.begin(9600); irrecv.enableIRIn(); // Запускаем приём Serial.println("=== ИК-ПРИЁМНИК: РЕЖИМ ДЕКОДИНГА ==="); Serial.println("Нажимайте кнопки на пульте"); } void loop() { if (irrecv.decode(&results)) { // Если получили сигнал Serial.print("Код: 0x"); Serial.print(results.value, HEX); // Выводим в 16-ричном виде Serial.print(" | Длина: "); Serial.print(results.bits); Serial.print(" бит | Протокол: "); // Определяем тип протокола switch (results.decode_type) { case NEC: Serial.println("NEC"); break; case SONY: Serial.println("SONY"); break; case RC5: Serial.println("RC5"); break; case RC6: Serial.println("RC6"); break; default: Serial.println("Unknown"); } irrecv.resume(); // Готовимся к следующему сигналу } }
🔬 ФАКТ О БИБЛИОТЕКЕ: Библиотека IRremote автоматически определяет протокол (NEC, Sony, RC5, RC6 и др.) и возвращает код нажатой кнопки . Важно: при удержании кнопки многие пульты посылают специальный "repeat code" (0xFFFFFFFF в библиотеке).
📊 ЗАПИШИТЕ СВОИ КОДЫ: Для дальнейших экспериментов вам понадобятся коды ваших кнопок. Обычно они выглядят как 0xFF00FF00. Рекомендую составить таблицу:

// ЭКСПЕРИМЕНТ 2: Управление светодиодом с пульта // Одна кнопка включает/выключает светодиод #include #define IR_PIN 11 #define LED_PIN 13 IRrecv irrecv(IR_PIN); decode_results results; // Код кнопки, которой будем управлять (замените на свой!) #define BUTTON_POWER 0xFF00FF00 bool ledState = false; void setup() { pinMode(LED_PIN, OUTPUT); digitalWrite(LED_PIN, LOW); Serial.begin(9600); irrecv.enableIRIn(); } void loop() { if (irrecv.decode(&results)) { // Проверяем, нажата ли нужная кнопка if (results.value == BUTTON_POWER) { ledState = !ledState; // Переключаем состояние digitalWrite(LED_PIN, ledState ? HIGH : LOW); Serial.print("Светодиод: "); Serial.println(ledState ? "ВКЛ" : "ВЫКЛ"); } irrecv.resume(); } }

// ЭКСПЕРИМЕНТ 3: Управление RGB-светодиодом с пульта // Кнопки: 1-красный, 2-зелёный, 3-синий, 4-выкл #include #define IR_PIN 11 #define RED_PIN 9 #define GREEN_PIN 10 #define BLUE_PIN 6 IRrecv irrecv(IR_PIN); decode_results results; // Коды кнопок вашего пульта (замените на свои!) #define BUTTON_RED 0xFF00FF00 #define BUTTON_GREEN 0xFF807F00 #define BUTTON_BLUE 0xFF40BF00 #define BUTTON_OFF 0xFF20DF00 void setup() { pinMode(RED_PIN, OUTPUT); pinMode(GREEN_PIN, OUTPUT); pinMode(BLUE_PIN, OUTPUT); // Выключаем всё digitalWrite(RED_PIN, LOW); digitalWrite(GREEN_PIN, LOW); digitalWrite(BLUE_PIN, LOW); Serial.begin(9600); irrecv.enableIRIn(); } void loop() { if (irrecv.decode(&results)) { switch(results.value) { case BUTTON_RED: digitalWrite(RED_PIN, HIGH); digitalWrite(GREEN_PIN, LOW); digitalWrite(BLUE_PIN, LOW); Serial.println("Красный"); break; case BUTTON_GREEN: digitalWrite(RED_PIN, LOW); digitalWrite(GREEN_PIN, HIGH); digitalWrite(BLUE_PIN, LOW); Serial.println("Зелёный"); break; case BUTTON_BLUE: digitalWrite(RED_PIN, LOW); digitalWrite(GREEN_PIN, LOW); digitalWrite(BLUE_PIN, HIGH); Serial.println("Синий"); break; case BUTTON_OFF: digitalWrite(RED_PIN, LOW); digitalWrite(GREEN_PIN, LOW); digitalWrite(BLUE_PIN, LOW); Serial.println("Выключено"); break; } irrecv.resume(); } }
🎨 ГРАДАЦИИ ЦВЕТА: В этом эксперименте мы просто включаем/выключаем цвета. Но можно использовать analogWrite() для ШИМ-регулировки яркости и получать миллионы оттенков. Например, кнопками "+"/"-" можно менять яркость красного, как в эксперименте 5 .

// ЭКСПЕРИМЕНТ 4: Управление сервоприводом с пульта // Кнопки: влево/вправо, центр #include #include #define IR_PIN 11 #define SERVO_PIN 9 IRrecv irrecv(IR_PIN); decode_results results; Servo myServo; int angle = 90; // Начальное положение // Коды кнопок (замените на свои!) #define BUTTON_LEFT 0xFF00FF00 #define BUTTON_RIGHT 0xFF807F00 #define BUTTON_CENTER 0xFF40BF00 void setup() { myServo.attach(SERVO_PIN); myServo.write(angle); Serial.begin(9600); irrecv.enableIRIn(); } void loop() { if (irrecv.decode(&results)) { // Обработка удержания: код повтора 0xFFFFFFFF if (results.value == 0xFFFFFFFF) { // Используем предыдущую команду (хранить в отдельной переменной) // Для простоты игнорируем } else { switch(results.value) { case BUTTON_LEFT: angle -= 10; if (angle < 0) angle = 0; Serial.print("Влево: "); Serial.println(angle); break; case BUTTON_RIGHT: angle += 10; if (angle > 180) angle = 180; Serial.print("Вправо: "); Serial.println(angle); break; case BUTTON_CENTER: angle = 90; Serial.println("Центр"); break; } myServo.write(angle); } irrecv.resume(); } }
🔄 ОБРАБОТКА УДЕРЖАНИЯ: Код 0xFFFFFFFF означает "повтор последней команды". В этом примере мы его игнорируем, но можно сделать так, чтобы при удержании сервопривод продолжал двигаться. Для этого нужно запоминать последнюю нажатую кнопку .

// ЭКСПЕРИМЕНТ 5: Диммер с ИК-пульта // Кнопки: громче = ярче, тише = тусклее #include #define IR_PIN 11 #define LED_PWM 9 IRrecv irrecv(IR_PIN); decode_results results; #define BUTTON_UP 0xFF00FF00 // Громче #define BUTTON_DOWN 0xFF807F00 // Тише int brightness = 0; bool lastCommandWasRepeat = false; unsigned long lastCommand = 0; void setup() { pinMode(LED_PWM, OUTPUT); analogWrite(LED_PWM, brightness); Serial.begin(9600); irrecv.enableIRIn(); } void loop() { if (irrecv.decode(&results)) { if (results.value == 0xFFFFFFFF && lastCommand != 0) { // Повтор последней команды if (lastCommand == BUTTON_UP) { brightness += 5; if (brightness > 255) brightness = 255; analogWrite(LED_PWM, brightness); Serial.print("Яркость: "); Serial.println(brightness); } else if (lastCommand == BUTTON_DOWN) { brightness -= 5; if (brightness < 0) brightness = 0; analogWrite(LED_PWM, brightness); Serial.print("Яркость: "); Serial.println(brightness); } } else { // Новая команда if (results.value == BUTTON_UP) { brightness += 15; if (brightness > 255) brightness = 255; analogWrite(LED_PWM, brightness); Serial.print("Яркость: "); Serial.println(brightness); lastCommand = BUTTON_UP; } else if (results.value == BUTTON_DOWN) { brightness -= 15; if (brightness < 0) brightness = 0; analogWrite(LED_PWM, brightness); Serial.print("Яркость: "); Serial.println(brightness); lastCommand = BUTTON_DOWN; } } irrecv.resume(); } }
💡 ПЛАВНАЯ РЕГУЛИРОВКА: Здесь мы используем ШИМ (analogWrite) для изменения яркости. При удержании кнопки (код 0xFFFFFFFF) продолжаем менять яркость в ту же сторону. Так работают настоящие диммеры и пульты для светодиодных лент .

// ЭКСПЕРИМЕНТ 6: ИК-пульт + LCD-дисплей // Показывает код нажатой кнопки и имя #include #include #include #define IR_PIN 11 IRrecv irrecv(IR_PIN); decode_results results; LiquidCrystal_I2C lcd(0x27, 16, 2); void setup() { lcd.init(); lcd.backlight(); lcd.setCursor(0, 0); lcd.print("IR Remote"); lcd.setCursor(0, 1); lcd.print("Press any key"); Serial.begin(9600); irrecv.enableIRIn(); delay(2000); lcd.clear(); } void loop() { if (irrecv.decode(&results)) { lcd.setCursor(0, 0); lcd.print("Code: "); lcd.setCursor(6, 0); // Выводим код в 16-ричном формате char buf[9]; sprintf(buf, "%08lX", results.value); lcd.print(buf); lcd.setCursor(0, 1); lcd.print(" "); // Очистка второй строки lcd.setCursor(0, 1); // Определяем популярные кнопки switch(results.value) { case 0xFF00FF00: lcd.print("BUTTON: POWER"); break; case 0xFF807F00: lcd.print("BUTTON: VOL+"); break; case 0xFF40BF00: lcd.print("BUTTON: VOL-"); break; case 0xFF20DF00: lcd.print("BUTTON: CH+"); break; case 0xFFA05F00: lcd.print("BUTTON: CH-"); break; default: if (results.value == 0xFFFFFFFF) { lcd.print("REPEAT"); } else { lcd.print("UNKNOWN"); } } irrecv.resume(); } }
📟 ФОРМАТИРОВАНИЕ: sprintf(buf, "%08lX", results.value) выводит 32-битное число в 16-ричном виде с ведущими нулями (8 символов). Это удобно для отладки и сравнения кодов .

// ЭКСПЕРИМЕНТ 7: Управление реле с ИК-пульта // Включаем/выключаем нагрузку (лампу, вентилятор) #include #define IR_PIN 11 #define RELAY_PIN 8 IRrecv irrecv(IR_PIN); decode_results results; #define BUTTON_POWER 0xFF00FF00 // Кнопка включения bool relayState = false; unsigned long lastPressTime = 0; const unsigned long debounceDelay = 300; // Защита от дребезга void setup() { pinMode(RELAY_PIN, OUTPUT); digitalWrite(RELAY_PIN, LOW); // Нагрузка выключена Serial.begin(9600); irrecv.enableIRIn(); } void loop() { if (irrecv.decode(&results)) { // Игнорируем повторные сигналы в течение 300 мс if (results.value != 0xFFFFFFFF && millis() - lastPressTime > debounceDelay) { if (results.value == BUTTON_POWER) { relayState = !relayState; digitalWrite(RELAY_PIN, relayState ? HIGH : LOW); Serial.print("Реле: "); Serial.println(relayState ? "ВКЛ" : "ВЫКЛ"); lastPressTime = millis(); } } irrecv.resume(); } }
