📱 ИК-ПРИЁМНИК VS1838 · 7 ИНЖЕНЕРНЫХ ЭКСПЕРИМЕНТОВ IRremote 4.x

Принцип работы: инфракрасный пульт модулирует сигнал частотой 38 кГц. Приёмник VS1838 демодулирует его и выдаёт цифровой сигнал, который Arduino расшифровывает с помощью библиотеки IRremote.

📚 ЧТО НОВОГО В IRREMOTE 4.x

IrReceiver.decode() — главная функция для получения данных

📁 Структура IrReceiver.decodedIRData:

{
  decodedRawData: 0xFF00FF00,  // 32-битный сырой код
  protocol: NEC,                // Протокол (NEC, SONY, RC5...)
  address: 0x00,                // Адрес устройства (для NEC)
  command: 0x45,                 // Команда (код кнопки)
  flags: 0x00                    // Флаги (повтор, ошибки)
}
🔬 ФАКТ О БИБЛИОТЕКЕ: В версии 4.x библиотека автоматически определяет протокол (NEC, Sony, RC5, RC6, Samsung и др.) и раскладывает код на address и command. Для NEC-пультов команда — это код нажатой кнопки, который обычно занимает 1 байт (0x00-0xFF).
🔌 БАЗОВОЕ ПОДКЛЮЧЕНИЕ (все эксперименты)
VCC → 5V (питание)
GND → GND (земля)
OUT → D11 (сигнал)
ℹ️ ВАЖНО ПРО РАСПИНОВКУ: У разных моделей ИК-приёмников распиновка может отличаться! Обычно:
Если смотреть на приёмник выпуклой стороной к себе: слева — OUT, посередине — GND, справа — VCC.
Всегда сверяйтесь с даташитом или маркировкой на корпусе.

// ЭКСПЕРИМЕНТ 1: Чтение кодов ИК-пульта (IRremote 4.x) // Установите библиотеку IRremote через менеджер библиотек #include // Обратите внимание на .hpp! #define IR_RECEIVE_PIN 11 // Пин подключения OUT void setup() { Serial.begin(115200); // Высокая скорость для отладки IrReceiver.begin(IR_RECEIVE_PIN); // Запускаем приёмник Serial.println(F("=== ИК-ПРИЁМНИК: РЕЖИМ ДЕКОДИНГА ===")); Serial.println(F("Нажимайте кнопки на пульте")); } void loop() { if (IrReceiver.decode()) { // Если получили сигнал // Выводим сырой код кнопки Serial.print(F("Код: 0x")); Serial.println(IrReceiver.decodedIRData.decodedRawData, HEX); // Дополнительная информация (для продвинутых) Serial.print(F("Протокол: ")); Serial.println(getProtocolString(IrReceiver.decodedIRData.protocol)); if (IrReceiver.decodedIRData.flags & IRDATA_FLAGS_IS_REPEAT) { Serial.println(F("ПОВТОР (удержание)")); } IrReceiver.resume(); // Готовимся к следующему сигналу } }
🔬 ФАКТ О КОДЕ: IrReceiver.decodedIRData.decodedRawData хранит 32-битное значение кнопки. Для NEC-пультов это обычно 4 байта: адрес (8 бит), инвертированный адрес (8 бит), команда (8 бит), инвертированная команда (8 бит). Но библиотека уже всё расшифровала — команду можно получить через IrReceiver.decodedIRData.command.

// ЭКСПЕРИМЕНТ 2: Управление светодиодом с пульта // Одна кнопка включает/выключает светодиод #include #define IR_RECEIVE_PIN 11 #define LED_PIN 13 bool ledState = false; void setup() { pinMode(LED_PIN, OUTPUT); digitalWrite(LED_PIN, LOW); Serial.begin(115200); IrReceiver.begin(IR_RECEIVE_PIN); Serial.println(F("=== УПРАВЛЕНИЕ СВЕТОДИОДОМ ===")); } void loop() { if (IrReceiver.decode()) { // Игнорируем повторы if (!(IrReceiver.decodedIRData.flags & IRDATA_FLAGS_IS_REPEAT)) { // Проверяем код кнопки (замените на свой!) if (IrReceiver.decodedIRData.decodedRawData == 0xFF00FF00) { ledState = !ledState; digitalWrite(LED_PIN, ledState ? HIGH : LOW); Serial.print(F("Светодиод: ")); Serial.println(ledState ? F("ВКЛ") : F("ВЫКЛ")); } } IrReceiver.resume(); } }
💡 ФЛАГ ПОВТОРА: IRDATA_FLAGS_IS_REPEAT позволяет отличить новое нажатие от удержания кнопки. Без этой проверки светодиод будет мигать при каждом повторе, а нам нужно переключение только один раз.

// ЭКСПЕРИМЕНТ 3: RGB-управление с пульта // Кнопки 1,2,3,4 управляют цветами #include #define IR_RECEIVE_PIN 11 #define RED_PIN 9 #define GREEN_PIN 10 #define BLUE_PIN 6 // Коды кнопок (замените на свои!) #define BUTTON_RED 0xFF00FF00 #define BUTTON_GREEN 0xFF807F00 #define BUTTON_BLUE 0xFF40BF00 #define BUTTON_OFF 0xFF20DF00 void setup() { pinMode(RED_PIN, OUTPUT); pinMode(GREEN_PIN, OUTPUT); pinMode(BLUE_PIN, OUTPUT); // Выключаем всё digitalWrite(RED_PIN, LOW); digitalWrite(GREEN_PIN, LOW); digitalWrite(BLUE_PIN, LOW); Serial.begin(115200); IrReceiver.begin(IR_RECEIVE_PIN); } void loop() { if (IrReceiver.decode()) { if (!(IrReceiver.decodedIRData.flags & IRDATA_FLAGS_IS_REPEAT)) { switch(IrReceiver.decodedIRData.decodedRawData) { case BUTTON_RED: digitalWrite(RED_PIN, HIGH); digitalWrite(GREEN_PIN, LOW); digitalWrite(BLUE_PIN, LOW); Serial.println(F("Красный")); break; case BUTTON_GREEN: digitalWrite(RED_PIN, LOW); digitalWrite(GREEN_PIN, HIGH); digitalWrite(BLUE_PIN, LOW); Serial.println(F("Зелёный")); break; case BUTTON_BLUE: digitalWrite(RED_PIN, LOW); digitalWrite(GREEN_PIN, LOW); digitalWrite(BLUE_PIN, HIGH); Serial.println(F("Синий")); break; case BUTTON_OFF: digitalWrite(RED_PIN, LOW); digitalWrite(GREEN_PIN, LOW); digitalWrite(BLUE_PIN, LOW); Serial.println(F("Выключено")); break; } } IrReceiver.resume(); } }
🎨 МИЛЛИОНЫ ЦВЕТОВ: Вместо включения/выключения можно использовать ШИМ (analogWrite) для плавной регулировки. Например, кнопками 5 и 6 менять яркость выбранного цвета. Так получается полноценный RGB-контроллер.

// ЭКСПЕРИМЕНТ 4: Управление сервоприводом с пульта #include #include #define IR_RECEIVE_PIN 11 #define SERVO_PIN 9 Servo myServo; int angle = 90; // Начальное положение // Коды кнопок (замените на свои!) #define BUTTON_LEFT 0xFF00FF00 #define BUTTON_RIGHT 0xFF807F00 #define BUTTON_CENTER 0xFF40BF00 void setup() { myServo.attach(SERVO_PIN); myServo.write(angle); Serial.begin(115200); IrReceiver.begin(IR_RECEIVE_PIN); } void loop() { if (IrReceiver.decode()) { if (!(IrReceiver.decodedIRData.flags & IRDATA_FLAGS_IS_REPEAT)) { switch(IrReceiver.decodedIRData.decodedRawData) { case BUTTON_LEFT: angle -= 10; if (angle < 0) angle = 0; Serial.print(F("Влево: ")); Serial.println(angle); break; case BUTTON_RIGHT: angle += 10; if (angle > 180) angle = 180; Serial.print(F("Вправо: ")); Serial.println(angle); break; case BUTTON_CENTER: angle = 90; Serial.println(F("Центр")); break; } myServo.write(angle); } IrReceiver.resume(); } }
🔄 ДЛЯ ПЛАВНОГО ДВИЖЕНИЯ: Если хотите, чтобы сервопривод двигался плавно при удержании кнопки, используйте флаг IRDATA_FLAGS_IS_REPEAT для постоянного изменения угла с небольшой задержкой.

// ЭКСПЕРИМЕНТ 5: Диммер с поддержкой удержания // Громче = ярче, тише = тусклее #include #define IR_RECEIVE_PIN 11 #define LED_PWM 9 #define BUTTON_UP 0xFF00FF00 // Громче (+) #define BUTTON_DOWN 0xFF807F00 // Тише (-) int brightness = 0; unsigned long lastAction = 0; void setup() { pinMode(LED_PWM, OUTPUT); analogWrite(LED_PWM, brightness); Serial.begin(115200); IrReceiver.begin(IR_RECEIVE_PIN); } void loop() { if (IrReceiver.decode()) { if (IrReceiver.decodedIRData.flags & IRDATA_FLAGS_IS_REPEAT) { // Удержание кнопки if (millis() - lastAction > 50) { // 50 мс между шагами if (IrReceiver.decodedIRData.decodedRawData == BUTTON_UP) { brightness += 5; if (brightness > 255) brightness = 255; analogWrite(LED_PWM, brightness); Serial.print(F("Яркость: ")); Serial.println(brightness); } else if (IrReceiver.decodedIRData.decodedRawData == BUTTON_DOWN) { brightness -= 5; if (brightness < 0) brightness = 0; analogWrite(LED_PWM, brightness); Serial.print(F("Яркость: ")); Serial.println(brightness); } lastAction = millis(); } } else { // Одиночное нажатие if (IrReceiver.decodedIRData.decodedRawData == BUTTON_UP) { brightness += 15; if (brightness > 255) brightness = 255; analogWrite(LED_PWM, brightness); Serial.print(F("Яркость: ")); Serial.println(brightness); } else if (IrReceiver.decodedIRData.decodedRawData == BUTTON_DOWN) { brightness -= 15; if (brightness < 0) brightness = 0; analogWrite(LED_PWM, brightness); Serial.print(F("Яркость: ")); Serial.println(brightness); } lastAction = millis(); } IrReceiver.resume(); } }
💡 УДЕРЖАНИЕ VS ПОВТОР: Флаг IRDATA_FLAGS_IS_REPEAT появляется при удержании кнопки. Частота повторов зависит от пульта (обычно ~40 мс). Мы добавляем собственную задержку 50 мс, чтобы яркость менялась плавно, а не скачками.

// ЭКСПЕРИМЕНТ 6: ИК-пульт + LCD-дисплей #include #include #include #define IR_RECEIVE_PIN 11 LiquidCrystal_I2C lcd(0x27, 16, 2); void setup() { lcd.init(); lcd.backlight(); lcd.setCursor(0, 0); lcd.print("IR Remote Ready"); Serial.begin(115200); IrReceiver.begin(IR_RECEIVE_PIN); delay(2000); lcd.clear(); lcd.setCursor(0, 0); lcd.print("Code:"); lcd.setCursor(0, 1); lcd.print("Cmd:"); } void loop() { if (IrReceiver.decode()) { // Вывод кода (верхняя строка) lcd.setCursor(6, 0); char buf[9]; sprintf(buf, "%08lX", IrReceiver.decodedIRData.decodedRawData); lcd.print(buf); // Вывод команды и протокола (нижняя строка) lcd.setCursor(5, 1); lcd.print(" "); // Очистка lcd.setCursor(5, 1); lcd.print(IrReceiver.decodedIRData.command, HEX); // Если есть адрес, покажем и его if (IrReceiver.decodedIRData.address != 0) { lcd.print(" A:"); lcd.print(IrReceiver.decodedIRData.address, HEX); } // Мигание при повторе if (IrReceiver.decodedIRData.flags & IRDATA_FLAGS_IS_REPEAT) { lcd.setCursor(15, 1); lcd.print("*"); } else { lcd.setCursor(15, 1); lcd.print(" "); } IrReceiver.resume(); } }
📟 ПОЛЕЗНЫЙ ПРИЁМ: sprintf(buf, "%08lX", value) форматирует 32-битное число в 8-символьный 16-ричный код с ведущими нулями. Это удобно для отладки и сравнения кодов.

// ЭКСПЕРИМЕНТ 7: Управление реле с ИК-пульта // Кнопка POWER включает/выключает нагрузку #include #define IR_RECEIVE_PIN 11 #define RELAY_PIN 8 #define BUTTON_POWER 0xFF00FF00 // Кнопка питания bool relayState = false; unsigned long lastPress = 0; const unsigned long debounceDelay = 300; void setup() { pinMode(RELAY_PIN, OUTPUT); digitalWrite(RELAY_PIN, LOW); Serial.begin(115200); IrReceiver.begin(IR_RECEIVE_PIN); } void loop() { if (IrReceiver.decode()) { // Игнорируем повторы if (!(IrReceiver.decodedIRData.flags & IRDATA_FLAGS_IS_REPEAT)) { // Защита от дребезга if (millis() - lastPress > debounceDelay) { if (IrReceiver.decodedIRData.decodedRawData == BUTTON_POWER) { relayState = !relayState; digitalWrite(RELAY_PIN, relayState ? HIGH : LOW); Serial.print(F("Реле: ")); Serial.println(relayState ? F("ВКЛ") : F("ВЫКЛ")); lastPress = millis(); } } } IrReceiver.resume(); } }
🏠 УМНЫЙ ДОМ: Точно так работают системы дистанционного управления светом. Добавив ESP8266, можно управлять нагрузкой через интернет, а ИК-пульт оставить как резервный канал.
